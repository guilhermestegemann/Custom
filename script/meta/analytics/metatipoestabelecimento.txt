AS
BEGIN

    NEW.ID = GEN_ID(GEN_METATIPOESTABELECIMENTO_ID,1);
END^

SET TERM ; ^

CREATE OR ALTER procedure set_bi_metatipoestabelecimento (
    distribuidor dom_str_18_nn,
    meta dom_id,
    tipoestabelecimento dom_id,
    valor dom_num_2d_nn,
    quantidade dom_num_3d_nn,
    pesobruto dom_num_3d_nn,
    volume dom_int_nn,
    positivacao dom_int_nn)
returns (
    id dom_id)
as
declare variable iddistribuidor dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    UPDATE OR INSERT INTO METATIPOESTABELECIMENTO(META, TIPOESTABELECIMENTO, VALOR, QUANTIDADE, PESOBRUTO, VOLUME, POSITIVACAO)
    VALUES (:IDMETA, :TIPOESTABELECIMENTO, :VALOR, :QUANTIDADE, :PESOBRUTO, :VOLUME, :POSITIVACAO)
    MATCHING (META, TIPOESTABELECIMENTO)
    RETURNING ID INTO :ID;

    SUSPEND;
END^

SET TERM ; ^

SET TERM ^ ;

CREATE OR ALTER procedure del_bi_metatipoestabelecimento (
    meta dom_id_isn,
    distribuidor dom_str_18_nn,
    tipoestabelecimento dom_id)
as
declare variable iddistribuidor dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    DELETE FROM METATIPOESTABELECIMENTO
    WHERE METATIPOESTABELECIMENTO.META = :IDMETA
    AND METATIPOESTABELECIMENTO.TIPOESTABELECIMENTO = :TIPOESTABELECIMENTO;
END^

SET TERM ; ^
