CREATE OR ALTER procedure set_bi_metacliente (
    distribuidor dom_str_18_nn,
    meta dom_id,
    cliente dom_str_18_nn,
    valor dom_num_2d_nn,
    quantidade dom_num_3d_nn,
    pesobruto dom_num_3d_nn,
    volume dom_int_nn)
returns (
    id dom_id)
as
declare variable iddistribuidor dom_id_isn;
declare variable idcliente dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_CLIENTE(:IDDISTRIBUIDOR, :CLIENTE) INTO :IDCLIENTE;
    IF (:IDCLIENTE IS NULL) THEN
        EXCEPTION EXCEPTION_CLIENTEINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    UPDATE OR INSERT INTO METACLIENTE(META, CLIENTE, VALOR, QUANTIDADE, PESOBRUTO, VOLUME)
    VALUES (:IDMETA, :IDCLIENTE, :VALOR, :QUANTIDADE, :PESOBRUTO, :VOLUME)
    MATCHING (META, CLIENTE)
    RETURNING ID INTO :ID;

    SUSPEND;
END^

SET TERM ; ^

SET TERM ^ ;

CREATE OR ALTER procedure del_bi_metacliente (
    meta dom_id_isn,
    distribuidor dom_str_18_nn,
    cliente dom_str_18_nn)
as
declare variable iddistribuidor dom_id_isn;
declare variable idcliente dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_CLIENTE(:IDDISTRIBUIDOR, :CLIENTE) INTO :IDCLIENTE;
    IF (:IDCLIENTE IS NULL) THEN
        EXCEPTION EXCEPTION_CLIENTEINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    DELETE FROM METACLIENTE
    WHERE METACLIENTE.META = :IDMETA
    AND METACLIENTE.CLIENTE = :IDCLIENTE;
END^

SET TERM ; ^
