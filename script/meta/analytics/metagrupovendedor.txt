    ID DOM_ID,
    META DOM_ID,
    GRUPO DOM_ID,
    VENDEDOR DOM_ID,
    VALOR DOM_NUM_2D_NN,
    QUANTIDADE DOM_NUM_3D_NN,
    PESOBRUTO DOM_NUM_3D_NN,
    VOLUME DOM_INT_NN,
    POSITIVACAO DOM_INT_NN);

CREATE SEQUENCE GEN_METAGRUPOVENDEDOR_ID;

SET TERM ^ ;

CREATE TRIGGER TRI_METAGRUPOVENDEDOR_BI FOR METAGRUPOVENDEDOR
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
  IF (NEW.ID IS NULL) THEN
    NEW.ID = GEN_ID(GEN_METAGRUPOVENDEDOR_ID,1);
END^

SET TERM ; ^

ALTER TABLE METAGRUPOVENDEDOR
ADD CONSTRAINT PK_METAGRUPOVENDEDOR
PRIMARY KEY (ID);

ADD CONSTRAINT FK_METAGRUPOVENDEDOR_META

ALTER TABLE METAGRUPOVENDEDOR

ALTER TABLE METAGRUPOVENDEDOR

ALTER TABLE METAGRUPOVENDEDOR

SET TERM ^ ;

CREATE OR ALTER procedure set_bi_metagrupovendedor (
    distribuidor dom_str_18_nn,
    meta dom_id,
    vendedor dom_str_18_nn,
    grupo dom_id,
    valor dom_num_2d_nn,
    quantidade dom_num_3d_nn,
    pesobruto dom_num_3d_nn,
    volume dom_int_nn,
    positivacao dom_int_nn)
returns (
    id dom_id)
as
declare variable iddistribuidor dom_id_isn;
declare variable idvendedor dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_VENDEDOR(:IDDISTRIBUIDOR, :VENDEDOR) INTO :IDVENDEDOR;
    IF (:IDVENDEDOR IS NULL) THEN
        EXCEPTION EXCEPTION_VENDEDORINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    UPDATE OR INSERT INTO METAGRUPOVENDEDOR(META, GRUPO, VENDEDOR, VALOR, QUANTIDADE, PESOBRUTO, VOLUME, POSITIVACAO)
    VALUES (:IDMETA, :GRUPO, :IDVENDEDOR,  :VALOR, :QUANTIDADE, :PESOBRUTO, :VOLUME, :POSITIVACAO)
    MATCHING (META, GRUPO, VENDEDOR)
    RETURNING ID INTO :ID;

    SUSPEND;
END^

SET TERM ; ^

SET TERM ^ ;

CREATE OR ALTER procedure del_bi_metagrupovendedor (
    meta dom_id_isn,
    distribuidor dom_str_18_nn,
    vendedor dom_str_18_nn,
    grupo dom_id)
as
declare variable iddistribuidor dom_id_isn;
declare variable idvendedor dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_VENDEDOR(:IDDISTRIBUIDOR, :VENDEDOR) INTO :IDVENDEDOR;
    IF (:IDVENDEDOR IS NULL) THEN
        EXCEPTION EXCEPTION_VENDEDORINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    DELETE FROM METAGRUPOVENDEDOR
    WHERE METAGRUPOVENDEDOR.META = :IDMETA
    AND METAGRUPOVENDEDOR.GRUPO = :GRUPO
    AND METAGRUPOVENDEDOR.VENDEDOR = :IDVENDEDOR;
END^

SET TERM ; ^
