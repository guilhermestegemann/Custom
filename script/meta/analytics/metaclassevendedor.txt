CREATE TABLE METACLASSEVENDEDOR (
    ID DOM_ID,
    META DOM_ID,
    CLASSE DOM_ID,
    VENDEDOR DOM_ID,
    VALOR DOM_NUM_2D_NN,
    QUANTIDADE DOM_NUM_3D_NN,
    PESOBRUTO DOM_NUM_3D_NN,
    VOLUME DOM_INT_NN,
    POSITIVACAO DOM_INT_NN);

CREATE SEQUENCE GEN_METACLASSEVENDEDOR_ID;

SET TERM ^ ;

CREATE TRIGGER TRI_METACLASSEVENDEDOR_BI FOR METACLASSEVENDEDOR
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
  IF (NEW.ID IS NULL) THEN
    NEW.ID = GEN_ID(GEN_METACLASSEVENDEDOR_ID,1);
END^

SET TERM ; ^

ALTER TABLE METACLASSEVENDEDOR
ADD CONSTRAINT PK_METACLASSEVENDEDOR
PRIMARY KEY (ID);

ALTER TABLE METACLASSEVENDEDOR
ADD CONSTRAINT FK_METACLASSEVENDEDOR_META
FOREIGN KEY (META)
REFERENCES META(ID);

ALTER TABLE METACLASSEVENDEDOR
ADD CONSTRAINT FK_METACLASSEVENDEDOR_CLASSE

ALTER TABLE METACLASSEVENDEDOR

ALTER TABLE METACLASSEVENDEDOR
ADD CONSTRAINT UNQ_METACLASSEVENDEDOR

SET TERM ^ ;

CREATE OR ALTER procedure set_bi_metaclassevendedor (
    distribuidor dom_str_18_nn,
    meta dom_id,
    vendedor dom_str_18_nn,
    classe dom_id,
    valor dom_num_2d_nn,
    quantidade dom_num_3d_nn,
    pesobruto dom_num_3d_nn,
    volume dom_int_nn,
    positivacao dom_int_nn)
returns (
    id dom_id)
as
declare variable iddistribuidor dom_id_isn;
declare variable idvendedor dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_VENDEDOR(:IDDISTRIBUIDOR, :VENDEDOR) INTO :IDVENDEDOR;
    IF (:IDVENDEDOR IS NULL) THEN
        EXCEPTION EXCEPTION_VENDEDORINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    UPDATE OR INSERT INTO METACLASSEVENDEDOR(META, CLASSE, VENDEDOR, VALOR, QUANTIDADE, PESOBRUTO, VOLUME, POSITIVACAO)
    VALUES (:IDMETA, :CLASSE, :IDVENDEDOR,  :VALOR, :QUANTIDADE, :PESOBRUTO, :VOLUME, :POSITIVACAO)
    MATCHING (META, CLASSE, VENDEDOR)
    RETURNING ID INTO :ID;

    SUSPEND;
END^

SET TERM ; ^

SET TERM ^ ;

CREATE OR ALTER procedure del_bi_metaclassevendedor (
    meta dom_id_isn,
    distribuidor dom_str_18_nn,
    vendedor dom_str_18_nn,
    classe dom_id)
as
declare variable iddistribuidor dom_id_isn;
declare variable idvendedor dom_id_isn;
declare variable idmeta dom_id_isn;
BEGIN
    SELECT ID FROM GET_DISTRIBUIDOR(:DISTRIBUIDOR) INTO :IDDISTRIBUIDOR;
    IF (:IDDISTRIBUIDOR IS NULL) THEN
        EXCEPTION EXCEPTION_DISTRIBUIDORINVALIDO;

    SELECT ID FROM GET_VENDEDOR(:IDDISTRIBUIDOR, :VENDEDOR) INTO :IDVENDEDOR;
    IF (:IDVENDEDOR IS NULL) THEN
        EXCEPTION EXCEPTION_VENDEDORINVALIDO;

    SELECT ID FROM GET_META(:IDDISTRIBUIDOR, :META) INTO :IDMETA;
    IF (:IDMETA IS NULL) THEN
        EXCEPTION EXCEPTION_METAINVALIDA;

    DELETE FROM METACLASSEVENDEDOR
    WHERE METACLASSEVENDEDOR.META = :IDMETA
    AND METACLASSEVENDEDOR.CLASSE = :CLASSE
    AND METACLASSEVENDEDOR.VENDEDOR = :IDVENDEDOR;
END^

SET TERM ; ^
