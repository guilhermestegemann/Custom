CREATE OR ALTER procedure cus_setfuncionario (
    codigo integer,
    nome varchar(60),
    cpf varchar(18),
    gerarpedidobloqueado varchar(1),
    comissaofixa varchar(1),
    utilizageolocalizacao varchar(1),
    afvliberarforarota varchar(1),
    enviarautomatico varchar(1),
    afvprontaentrega varchar(1),
    comissao numeric(8,2),
    exportar varchar(1),
    ativo varchar(1),
    recebemensagem varchar(1),
    gerarcomissao varchar(1),
    flex numeric(8,2),
    gerarflex varchar(1),
    utilizarota varchar(1),
    afvtemposincronizar integer,
    limiteflex numeric(15,2))
as
declare variable contfuncionario integer;
BEGIN
    SELECT COUNT(*)
    FROM FUNCIONARIO
    WHERE FUNCIONARIO.CODIGO = :CODIGO
    INTO :CONTFUNCIONARIO;
    IF (:CONTFUNCIONARIO = 0) THEN
        INSERT INTO FUNCIONARIO
        (
            CODIGO,
            NOME,
            CPF,
            GERARPEDIDOBLOQUEADO,
            COMISSAOFIXA,
            UTILIZAGEOLOCALIZACAO,
            AFVLIBERARFORAROTA,
            ENVIARAUTOMATICO,
            AFVPRONTAENTREGA,
            COMISSAO,
            EXPORTAR,
            ATIVO,
            RECEBEMENSAGEM,
            GERARCOMISSAO,
            FLEX,
            GERARFLEX,
            UTILIZAROTA,
            AFVTEMPOSINCRONIZAR,
            LIMITEFLEX
        ) VALUES
        (
            :CODIGO,
            :NOME,
            :CPF,
            :GERARPEDIDOBLOQUEADO,
            :COMISSAOFIXA,
            :UTILIZAGEOLOCALIZACAO,
            :AFVLIBERARFORAROTA,
            :ENVIARAUTOMATICO,
            :AFVPRONTAENTREGA,
            :COMISSAO,
            :EXPORTAR,
            :ATIVO,
            :RECEBEMENSAGEM,
            :GERARCOMISSAO,
            :FLEX,
            :GERARFLEX,
            :UTILIZAROTA,
            :AFVTEMPOSINCRONIZAR,
            :LIMITEFLEX
        );
END^

SET TERM ; ^
