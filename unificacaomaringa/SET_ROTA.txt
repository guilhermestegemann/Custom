CREATE OR ALTER procedure cus_setrota (
    codigo integer,
    nome varchar(30),
    tiporota varchar(1),
    funcionario integer,
    classificacaorota integer,
    filial integer)
as
BEGIN
    CODIGO = '100'||:CODIGO;
    INSERT INTO ROTA
    (
        CODIGO,
        NOME,
        TIPOROTA,
        FUNCIONARIO,
        CLASSIFICACAOROTA,
        FILIAL,
        REORDENAR
    ) VALUES
    (
        :CODIGO,
        :NOME,
        :TIPOROTA,
        :FUNCIONARIO,
        :CLASSIFICACAOROTA,
        :FILIAL,
        'N'
    );
END^
