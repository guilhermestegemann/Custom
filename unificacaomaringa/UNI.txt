--SQL CLIENTE

FROM CLIFOR C

--SQL CLIFORCONTATO
SELECT CLIFOR.CNPJ AS CLIFOR, CC.NUMERO, CC.NOME, CC.EMAIL, CC.ENVIARNFE, CC.ENVIARDANFE, CC.ENVIARBOLETO, CC.ENVIARPEDIDO, 2 AS FILIAL
FROM CLIFORCONTATO CC
INNER JOIN CLIFOR ON CLIFOR.CODIGO = CC.CLIFOR

--SQL FUNCIONARIOCLIFOR
SELECT FUNCIONARIOCLIFOR.FUNCIONARIO, CLIFOR.CNPJ AS CLIENTE
FROM FUNCIONARIOCLIFOR
INNER JOIN CLIFOR ON FUNCIONARIOCLIFOR.CLIFOR = CLIFOR.CODIGO
WHERE FUNCIONARIOCLIFOR.FUNCIONARIO IN (2089,2976,264)
AND CLIFOR.CNPJ IS NOT NULL

--SQL ROTA
SELECT CODIGO, NOME, TIPOROTA, FUNCIONARIO, CLASSIFICACAOROTA, 2 AS FILIAL
FROM ROTA
WHERE ROTA.FUNCIONARIO IN (2089,2976,264)

--SQL ROTACLIFOR
SELECT ROTACLIFOR.ROTA, CLIFOR.CNPJ, ROTACLIFOR.ORDEM
FROM ROTACLIFOR
INNER JOIN ROTA ON ROTA.CODIGO = ROTACLIFOR.ROTA
INNER JOIN CLIFOR ON CLIFOR.CODIGO = ROTACLIFOR.CLIFOR
WHERE ROTA.FUNCIONARIO IN (2089,2976,264)
