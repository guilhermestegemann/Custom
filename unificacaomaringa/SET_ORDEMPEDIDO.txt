CREATE OR ALTER procedure cus_setordempedido (
    idpedido char(36),
    numero integer,
    filial integer,
    ordem varchar(15),
    vcto date,
    valor numeric(8,2),
    enviar varchar(1),
    alterada varchar(1))
as
BEGIN
    INSERT INTO ORDEMPEDIDO
    (
        IDPEDIDO,
        NUMERO,
        FILIAL,
        ORDEM,
        VCTO,
        VALOR,
        ENVIAR,
        ALTERADA
    ) VALUES
    (
        CHAR_TO_UUID(:IDPEDIDO),
        :NUMERO,
        :FILIAL,
        :ORDEM,
        :VCTO,
        :VALOR,
        :ENVIAR,
        :ALTERADA
    );
END^

SET TERM ; ^
